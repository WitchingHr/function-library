Random number between 0 and another number:

function randomInt(max) {
    return Math.floor(Math.random() * max);
}

Squared number:

function squared(num) {
    return num * num;
}

Cubed number:

function cubed(num) {
    return num * num * num;
}

Factorial:

function factorial(num) {
    if (num < 0) return undefined;
    if (num === 0) return 1;
    let x = num - 1;
    while (x > 1) {
        num *= x;
        x--;
    }
    return num;
}
